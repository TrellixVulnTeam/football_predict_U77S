Create Table Competitions( 
    ID INT NOT NULL AUTO_INCREMENT, 
    Name varchar(100) NOT NULL, 
    DateCreated date NOT NULL, 
    DateEnd date NOT NULL, 
    Week INT NOT NULL,
    JoinCode varchar(10) NOT NULL,
    PRIMARY KEY(ID)
);

CREATE TABLE Users ( 
    ID INT NOT NULL AUTO_INCREMENT, 
    Name VARCHAR(100) NOT NULL, 
    WinCount INT NOT NULL, 
    DisplayName VARCHAR(50),
    PRIMARY KEY(ID)
);

CREATE TABLE Competing (

    Pick1 INT, 
    Pick2 INT, 
    Pick3 INT, 
    Pick4 INT, 
    Pick5 INT, 
    Pick6 INT, 
    Pick7 INT, 
    Pick8 INT, 
    Pick9 INT, 
    Pick10 INT, 
    Pick11 INT, 
    Pick12 INT, 
    Pick13 INT, 
    Pick14 INT, 
    Pick15 INT, 
    Pick16 INT
);

CREATE TABLE Games (
    Week INT NOT NULL,
    Team VARCHAR(50),
    PickSlot INT NOT NULL,
    Home INT NOT NULL
);
